文件中存放100亿个数字， 范围是0-100亿，请设计一个算法，从文件里存放的数字数组中，找出重复的数字

举例：文件存放内容nums：6, 7, 9, 10, 101, 1, 6, 3, 7……，程序输出结果：6，7，……
nums：6, 7, 9, 10, 101, 1, 6, 3, 7…

#include<iostream>

#include<bits/stdc++.h>
int main(){

  vector<int> flags(100e,0);

  vector<int> ans;

  // (nums.begin(),nums.begin()+10e);

  O(nlogn)

  for(auto i:nums){

    if(flags[i]){

      ans.push_back(i);

    }

    flags[i]=1;

  }

  return ans;

}
